local TweenService = game:GetService("TweenService")
local Players = game:GetService("Players")
local player = Players.LocalPlayer
local PlayerGui = player:WaitForChild("PlayerGui")

local gui = Instance.new("ScreenGui")
gui.Name = "BrutonIntro"
gui.IgnoreGuiInset = true
gui.ResetOnSpawn = false
gui.Parent = PlayerGui

local blurFrame = Instance.new("Frame")
blurFrame.Size = UDim2.new(1,0,1,0)
blurFrame.BackgroundColor3 = Color3.fromRGB(120,0,0)
blurFrame.BackgroundTransparency = 0.4
blurFrame.BorderSizePixel = 0
blurFrame.Parent = gui

local blurEffect = Instance.new("BlurEffect")
blurEffect.Size = 24
blurEffect.Parent = game.Lighting

for i = 1, 50 do
    local drop = Instance.new("TextLabel")
    drop.Size = UDim2.new(0,20,0,20)
    drop.Position = UDim2.new(math.random(),0,math.random(),0)
    drop.BackgroundTransparency = 1
    drop.Text = ""
    drop.TextScaled = true
    drop.TextTransparency = 0.6
    drop.Parent = gui

    task.spawn(function()
        while gui.Parent do
            TweenService:Create(drop, TweenInfo.new(math.random(2,4), Enum.EasingStyle.Linear), {
                Position = UDim2.new(drop.Position.X.Scale,0,1.1,0)
            }):Play()
            task.wait(math.random(2,4))
            drop.Position = UDim2.new(math.random(),0,-0.1,0)
        end
    end)
end

local barContainer = Instance.new("Frame")
barContainer.Size = UDim2.new(0.6,0,0.05,0)
barContainer.Position = UDim2.new(0.2,0,0.8,0)
barContainer.BackgroundColor3 = Color3.fromRGB(60,0,0)
barContainer.BackgroundTransparency = 0.5
barContainer.BorderSizePixel = 0
barContainer.ClipsDescendants = true
barContainer.Parent = gui
barContainer.AnchorPoint = Vector2.new(0,0)

local UICorner = Instance.new("UICorner")
UICorner.CornerRadius = UDim.new(0,15)
UICorner.Parent = barContainer

local barFill = Instance.new("Frame")
barFill.Size = UDim2.new(0,0,1,0)
barFill.BackgroundColor3 = Color3.fromRGB(255,0,0)
barFill.BorderSizePixel = 0
barFill.Parent = barContainer

local barGlow = Instance.new("UIGradient")
barGlow.Color = ColorSequence.new{
    ColorSequenceKeypoint.new(0, Color3.fromRGB(255,30,30)),
    ColorSequenceKeypoint.new(1, Color3.fromRGB(255,0,0))
}
barGlow.Rotation = 0
barGlow.Parent = barFill

local title = Instance.new("TextLabel")
title.Size = UDim2.new(1,0,0.2,0)
title.Position = UDim2.new(0,0,0.35,0)
title.BackgroundTransparency = 1
title.Text = "Sorry"
title.TextColor3 = Color3.fromRGB(255,0,0)
title.Font = Enum.Font.GothamBold
title.TextScaled = true
title.Parent = gui

spawn(function()
    while gui.Parent do
        TweenService:Create(title, TweenInfo.new(1.5, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            TextColor3 = Color3.fromRGB(160,0,0)
        }):Play()
        task.wait(1.5)
        TweenService:Create(title, TweenInfo.new(1.5, Enum.EasingStyle.Quad, Enum.EasingDirection.Out), {
            TextColor3 = Color3.fromRGB(255,40,40)
        }):Play()
        task.wait(1.5)
    end
end)

TweenService:Create(barFill, TweenInfo.new(6, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {
    Size = UDim2.new(1,0,1,0)
}):Play()

task.wait(6.5)
TweenService:Create(gui, TweenInfo.new(1, Enum.EasingStyle.Sine, Enum.EasingDirection.Out), {
    BackgroundTransparency = 1
}):Play()
TweenService:Create(blurFrame, TweenInfo.new(1, Enum.EasingStyle.Sine), {BackgroundTransparency = 1}):Play()
TweenService:Create(title, TweenInfo.new(1, Enum.EasingStyle.Sine), {TextTransparency = 1}):Play()
TweenService:Create(barContainer, TweenInfo.new(1, Enum.EasingStyle.Sine), {BackgroundTransparency = 1}):Play()
task.wait(1.2)
blurEffect:Destroy()
gui:Destroy()
